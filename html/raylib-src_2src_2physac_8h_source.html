1 /**********************************************************************************************
2 *
3 * Physac v1.0 - 2D Physics library for videogames
4 *
5 * DESCRIPTION:
6 *
7 * Physac is a small 2D physics engine written in pure C. The engine uses a fixed time-step thread loop
8 * to simluate physics. A physics step contains the following phases: get collision information,
9 * apply dynamics, collision solving and position correction. It uses a very simple struct for physic
10 * bodies with a position vector to be used in any 3D rendering API.
11 *
12 * CONFIGURATION:
13 *
14 * #define PHYSAC_IMPLEMENTATION
15 * Generates the implementation of the library into the included file.
16 * If not defined, the library is in header only mode and can be included in other headers
17 * or source files without problems. But only ONE file should hold the implementation.
18 *
19 * #define PHYSAC_STATIC (defined by default)
20 * The generated implementation will stay private inside implementation file and all
21 * internal symbols and functions will only be visible inside that file.
22 *
23 * #define PHYSAC_NO_THREADS
24 * The generated implementation won't include pthread library and user must create a secondary thread to call PhysicsThread().
25 * It is so important that the thread where PhysicsThread() is called must not have v-sync or any other CPU limitation.
26 *
27 * #define PHYSAC_STANDALONE
28 * Avoid raylib.h header inclusion in this file. Data types defined on raylib are defined
29 * internally in the library and input management and drawing functions must be provided by
30 * the user (check library implementation for further details).
31 *
32 * #define PHYSAC_DEBUG
33 * Traces log messages when creating and destroying physics bodies and detects errors in physics
34 * calculations and reference exceptions; it is useful for debug purposes
35 *
36 * #define PHYSAC_MALLOC()
37 * #define PHYSAC_FREE()
38 * You can define your own malloc/free implementation replacing stdlib.h malloc()/free() functions.
39 * Otherwise it will include stdlib.h and use the C standard library malloc()/free() function.
40 *
41 *
42 * NOTE 1: Physac requires multi-threading, when InitPhysics() a second thread is created to manage physics calculations.
43 * NOTE 2: Physac requires static C library linkage to avoid dependency on MinGW DLL (-static -lpthread)
44 *
45 * Use the following code to compile:
46 * gcc -o $(NAME_PART).exe $(FILE_NAME) -s -static -lraylib -lpthread -lopengl32 -lgdi32 -lwinmm -std=c99

381 // Creates a new rectangle physics body with generic parameters
382 PHYSACDEF PhysicsBody CreatePhysicsBodyRectangle(Vector2 pos, float width, float height, float density)
383 {
384  PhysicsBody newBody = (PhysicsBody)PHYSAC_MALLOC(sizeof(PhysicsBodyData));
385  usedMemory += sizeof(PhysicsBodyData);
386 
387  int newId = FindAvailableBodyIndex();
388  if (newId != -1)
389  {
390  // Initialize new body with generic values
391  newBody->id = newId;
392  newBody->enabled = true;
393  newBody->position = pos;
394  newBody->velocity = (Vector2){ 0.0f };
395  newBody->force = (Vector2){ 0.0f };
396  newBody->angularVelocity = 0.0f;
397  newBody->torque = 0.0f;
398  newBody->orient = 0.0f;
399  newBody->shape.type = PHYSICS_POLYGON;
400  newBody->shape.body = newBody;
401  newBody->shape.radius = 0.0f;
402  newBody->shape.transform = Mat2Radians(0.0f);
403  newBody->shape.vertexData = CreateRectanglePolygon(pos, (Vector2){ width, height });
404 
405  // Calculate centroid and moment of inertia
406  Vector2 center = { 0.0f, 0.0f };
407  float area = 0.0f;
408  float inertia = 0.0f;
409 
410  for (int i = 0; i < newBody->shape.vertexData.vertexCount; i++)
411  {
412  // Triangle vertices, third vertex implied as (0, 0)
413  Vector2 p1 = newBody->shape.vertexData.positions[i];
414  int nextIndex = (((i + 1) < newBody->shape.vertexData.vertexCount) ? (i + 1) : 0);
415  Vector2 p2 = newBody->shape.vertexData.positions[nextIndex];
416 
417  float D = MathCrossVector2(p1, p2);
418  float triangleArea = D/2;
419 
420  area += triangleArea;
421 
422  // Use area to weight the centroid average, not just vertex position
423  center.x += triangleArea*PHYSAC_K*(p1.x + p2.x);
424  center.y += triangleArea*PHYSAC_K*(p1.y + p2.y);
425 
426  float intx2 = p1.x*p1.x + p2.x*p1.x + p2.x*p2.x;
427  float inty2 = p1.y*p1.y + p2.y*p1.y + p2.y*p2.y;
428  inertia += (0.25f*PHYSAC_K*D)*(intx2 + inty2);
429  }
430 
431  center.x *= 1.0f/area;
432  center.y *= 1.0f/area;
433 
434  // Translate vertices to centroid (make the centroid (0, 0) for the polygon in model space)
435  // Note: this is not really necessary
436  for (int i = 0; i < newBody->shape.vertexData.vertexCount; i++)
437  {
438  newBody->shape.vertexData.positions[i].x -= center.x;
439  newBody->shape.vertexData.positions[i].y -= center.y;
440  }
441 
442  newBody->mass = density*area;
443  newBody->inverseMass = ((newBody->mass != 0.0f) ? 1.0f/newBody->mass : 0.0f);
444  newBody->inertia = density*inertia;
445  newBody->inverseInertia = ((newBody->inertia != 0.0f) ? 1.0f/newBody->inertia : 0.0f);
446  newBody->staticFriction = 0.4f;
447  newBody->dynamicFriction = 0.2f;
448  newBody->restitution = 0.0f;
449  newBody->useGravity = true;
450  newBody->isGrounded = false;
451  newBody->freezeOrient = false;
452 
453  // Add new body to bodies pointers array and update bodies count
454  bodies[physicsBodiesCount] = newBody;
455  physicsBodiesCount++;
456 
457  #if defined(PHYSAC_DEBUG)
458  printf("[PHYSAC] created polygon physics body id %i\n", newBody->id);
459  #endif
460  }
461  #if defined(PHYSAC_DEBUG)
462  else printf("[PHYSAC] new physics body creation failed because there is any available id to use\n");
463  #endif
464 
465  return newBody;
466 }

566 // Shatters a polygon shape physics body to little physics bodies with explosion force
567 PHYSACDEF void PhysicsShatter(PhysicsBody body, Vector2 position, float force)
568 {
569  if (body != NULL)
570  {
571  if (body->shape.type == PHYSICS_POLYGON)
572  {
573  PolygonData vertexData = body->shape.vertexData;
574  bool collision = false;
575 
576  for (int i = 0; i < vertexData.vertexCount; i++)
577  {
578  Vector2 positionA = body->position;
579  Vector2 positionB = Mat2MultiplyVector2(body->shape.transform, Vector2Add(body->position, vertexData.positions[i]));
580  int nextIndex = (((i + 1) < vertexData.vertexCount) ? (i + 1) : 0);
581  Vector2 positionC = Mat2MultiplyVector2(body->shape.transform, Vector2Add(body->position, vertexData.positions[nextIndex]));
582 
583  // Check collision between each triangle
584  float alpha = ((positionB.y - positionC.y)*(position.x - positionC.x) + (positionC.x - positionB.x)*(position.y - positionC.y))/
585  ((positionB.y - positionC.y)*(positionA.x - positionC.x) + (positionC.x - positionB.x)*(positionA.y - positionC.y));
586 
587  float beta = ((positionC.y - positionA.y)*(position.x - positionC.x) + (positionA.x - positionC.x)*(position.y - positionC.y))/
588  ((positionB.y - positionC.y)*(positionA.x - positionC.x) + (positionC.x - positionB.x)*(positionA.y - positionC.y));
589 
590  float gamma = 1.0f - alpha - beta;
591 
592  if ((alpha > 0.0f) && (beta > 0.0f) & (gamma > 0.0f))
593  {
594  collision = true;
595  break;
596  }
597  }
598 
599  if (collision)
600  {
601  int count = vertexData.vertexCount;
602  Vector2 bodyPos = body->position;
603  Vector2 *vertices = (Vector2*)malloc(sizeof(Vector2) * count);
604  Matrix2x2 trans = body->shape.transform;
605  for (int i = 0; i < count; i++) vertices[i] = vertexData.positions[i];
606 
607  // Destroy shattered physics body
608  DestroyPhysicsBody(body);
609 
610  for (int i = 0; i < count; i++)
611  {
612  int nextIndex = (((i + 1) < count) ? (i + 1) : 0);
613  Vector2 center = TriangleBarycenter(vertices[i], vertices[nextIndex], PHYSAC_VECTOR_ZERO);
614  center = Vector2Add(bodyPos, center);
615  Vector2 offset = Vector2Subtract(center, bodyPos);
616 
617  PhysicsBody newBody = CreatePhysicsBodyPolygon(center, 10, 3, 10); // Create polygon physics body with relevant values
618 
619  PolygonData newData = { 0 };
620  newData.vertexCount = 3;
621 
622  newData.positions[0] = Vector2Subtract(vertices[i], offset);
623  newData.positions[1] = Vector2Subtract(vertices[nextIndex], offset);
624  newData.positions[2] = Vector2Subtract(position, center);
625 
626  // Separate vertices to avoid unnecessary physics collisions
627  newData.positions[0].x *= 0.95f;
628  newData.positions[0].y *= 0.95f;
629  newData.positions[1].x *= 0.95f;
630  newData.positions[1].y *= 0.95f;
631  newData.positions[2].x *= 0.95f;
632  newData.positions[2].y *= 0.95f;
633 
634  // Calculate polygon faces normals
635  for (int j = 0; j < newData.vertexCount; j++)
636  {
637  int nextVertex = (((j + 1) < newData.vertexCount) ? (j + 1) : 0);
638  Vector2 face = Vector2Subtract(newData.positions[nextVertex], newData.positions[j]);
639 
640  newData.normals[j] = (Vector2){ face.y, -face.x };
641  MathNormalize(&newData.normals[j]);
642  }
643 
644  // Apply computed vertex data to new physics body shape
645  newBody->shape.vertexData = newData;
646  newBody->shape.transform = trans;
647 
648  // Calculate centroid and moment of inertia
649  center = PHYSAC_VECTOR_ZERO;
650  float area = 0.0f;
651  float inertia = 0.0f;
652 
653  for (int j = 0; j < newBody->shape.vertexData.vertexCount; j++)
654  {
655  // Triangle vertices, third vertex implied as (0, 0)
656  Vector2 p1 = newBody->shape.vertexData.positions[j];
657  int nextVertex = (((j + 1) < newBody->shape.vertexData.vertexCount) ? (j + 1) : 0);
658  Vector2 p2 = newBody->shape.vertexData.positions[nextVertex];
659 
660  float D = MathCrossVector2(p1, p2);
661  float triangleArea = D/2;
662 
663  area += triangleArea;
664 
665  // Use area to weight the centroid average, not just vertex position
666  center.x += triangleArea*PHYSAC_K*(p1.x + p2.x);
667  center.y += triangleArea*PHYSAC_K*(p1.y + p2.y);
668 
669  float intx2 = p1.x*p1.x + p2.x*p1.x + p2.x*p2.x;
670  float inty2 = p1.y*p1.y + p2.y*p1.y + p2.y*p2.y;
671  inertia += (0.25f*PHYSAC_K*D)*(intx2 + inty2);
672  }
673 
674  center.x *= 1.0f/area;
675  center.y *= 1.0f/area;
676 
677  newBody->mass = area;
678  newBody->inverseMass = ((newBody->mass != 0.0f) ? 1.0f/newBody->mass : 0.0f);
679  newBody->inertia = inertia;
680  newBody->inverseInertia = ((newBody->inertia != 0.0f) ? 1.0f/newBody->inertia : 0.0f);
681 
682  // Calculate explosion force direction
683  Vector2 pointA = newBody->position;
684  Vector2 pointB = Vector2Subtract(newData.positions[1], newData.positions[0]);
685  pointB.x /= 2.0f;
686  pointB.y /= 2.0f;
687  Vector2 forceDirection = Vector2Subtract(Vector2Add(pointA, Vector2Add(newData.positions[0], pointB)), newBody->position);
688  MathNormalize(&forceDirection);
689  forceDirection.x *= force;
690  forceDirection.y *= force;
691 
692  // Apply force to new physics body
693  PhysicsAddForce(newBody, forceDirection);
694  }
695 
696  free(vertices);
697  }
698  }
699  }
700  #if defined(PHYSAC_DEBUG)
701  else printf("[PHYSAC] error when trying to shatter a null reference physics body");
702  #endif
703 }

1037 // Physics steps calculations (dynamics, collisions and position corrections)
1038 static void PhysicsStep(void)
1039 {
1040  // Update current steps count
1041  stepsCount++;
1042 
1043  // Clear previous generated collisions information
1044  for (int i = physicsManifoldsCount - 1; i >= 0; i--)
1045  {
1046  PhysicsManifold manifold = contacts[i];
1047  if (manifold != NULL) DestroyPhysicsManifold(manifold);
1048  }
1049 
1050  // Reset physics bodies grounded state
1051  for (int i = 0; i < physicsBodiesCount; i++)
1052  {
1053  PhysicsBody body = bodies[i];
1054  body->isGrounded = false;
1055  }
1056 
1057  // Generate new collision information
1058  for (int i = 0; i < physicsBodiesCount; i++)
1059  {
1060  PhysicsBody bodyA = bodies[i];
1061 
1062  if (bodyA != NULL)
1063  {
1064  for (int j = i + 1; j < physicsBodiesCount; j++)
1065  {
1066  PhysicsBody bodyB = bodies[j];
1067 
1068  if (bodyB != NULL)
1069  {
1070  if ((bodyA->inverseMass == 0) && (bodyB->inverseMass == 0)) continue;
1071 
1072  PhysicsManifold manifold = CreatePhysicsManifold(bodyA, bodyB);
1073  SolvePhysicsManifold(manifold);
1074 
1075  if (manifold->contactsCount > 0)
1076  {
1077  // Create a new manifold with same information as previously solved manifold and add it to the manifolds pool last slot
1078  PhysicsManifold newManifold = CreatePhysicsManifold(bodyA, bodyB);
1079  newManifold->penetration = manifold->penetration;
1080  newManifold->normal = manifold->normal;
1081  newManifold->contacts[0] = manifold->contacts[0];
1082  newManifold->contacts[1] = manifold->contacts[1];
1083  newManifold->contactsCount = manifold->contactsCount;
1084  newManifold->restitution = manifold->restitution;
1085  newManifold->dynamicFriction = manifold->dynamicFriction;
1086  newManifold->staticFriction = manifold->staticFriction;
1087  }
1088  }
1089  }
1090  }
1091  }
1092 
1093  // Integrate forces to physics bodies
1094  for (int i = 0; i < physicsBodiesCount; i++)
1095  {
1096  PhysicsBody body = bodies[i];
1097  if (body != NULL) IntegratePhysicsForces(body);
1098  }
1099 
1100  // Initialize physics manifolds to solve collisions
1101  for (int i = 0; i < physicsManifoldsCount; i++)
1102  {
1103  PhysicsManifold manifold = contacts[i];
1104  if (manifold != NULL) InitializePhysicsManifolds(manifold);
1105  }
1106 
1107  // Integrate physics collisions impulses to solve collisions
1108  for (int i = 0; i < PHYSAC_COLLISION_ITERATIONS; i++)
1109  {
1110  for (int j = 0; j < physicsManifoldsCount; j++)
1111  {
1112  PhysicsManifold manifold = contacts[i];
1113  if (manifold != NULL) IntegratePhysicsImpulses(manifold);
1114  }
1115  }
1116 
1117  // Integrate velocity to physics bodies
1118  for (int i = 0; i < physicsBodiesCount; i++)
1119  {
1120  PhysicsBody body = bodies[i];
1121  if (body != NULL) IntegratePhysicsVelocity(body);
1122  }
1123 
1124  // Correct physics bodies positions based on manifolds collision information
1125  for (int i = 0; i < physicsManifoldsCount; i++)
1126  {
1127  PhysicsManifold manifold = contacts[i];
1128  if (manifold != NULL) CorrectPhysicsPositions(manifold);
1129  }
1130 
1131  // Clear physics bodies forces
1132  for (int i = 0; i < physicsBodiesCount; i++)
1133  {
1134  PhysicsBody body = bodies[i];
1135  if (body != NULL)
1136  {
1137  body->force = PHYSAC_VECTOR_ZERO;
1138  body->torque = 0.0f;
1139  }
1140  }
1141 }

1311 // Solves collision between two circle shape physics bodies

1313 {

1316

1338  manifold->normal = (Vector2){ 1.0f, 0.0f };

1340  }
1341  else
1342  {

1344  manifold->normal = (Vector2){ normal.x/distance, normal.y/distance }; // Faster than using MathNormalize() due to sqrt is already performed
1345  manifold->contacts[0] = (Vector2){ manifold->normal.x*bodyA->shape.radius + bodyA->position.x, manifold->normal.y*bodyA->shape.radius + bodyA->position.y };

1349  if (!bodyA->isGrounded) bodyA->isGrounded = (manifold->normal.y < 0);

1352 // Solves collision between a circle to a polygon shape physics bodies
1353 static void SolveCircleToPolygon(PhysicsManifold manifold)
1354 {
1355  PhysicsBody bodyA = manifold->bodyA;
1356  PhysicsBody bodyB = manifold->bodyB;
1357 
1358  if ((bodyA == NULL) || (bodyB == NULL)) return;
1359 
1360  manifold->contactsCount = 0;
1361 
1362  // Transform circle center to polygon transform space
1363  Vector2 center = bodyA->position;
1364  center = Mat2MultiplyVector2(Mat2Transpose(bodyB->shape.transform), Vector2Subtract(center, bodyB->position));
1365 
1366  // Find edge with minimum penetration
1367  // It is the same concept as using support points in SolvePolygonToPolygon
1368  float separation = -PHYSAC_FLT_MAX;
1369  int faceNormal = 0;
1370  PolygonData vertexData = bodyB->shape.vertexData;
1371 
1372  for (int i = 0; i < vertexData.vertexCount; i++)
1373  {
1374  float currentSeparation = MathDot(vertexData.normals[i], Vector2Subtract(center, vertexData.positions[i]));
1375 
1376  if (currentSeparation > bodyA->shape.radius) return;
1377 
1378  if (currentSeparation > separation)
1379  {
1380  separation = currentSeparation;
1381  faceNormal = i;
1382  }
1383  }
1384 
1385  // Grab face's vertices
1386  Vector2 v1 = vertexData.positions[faceNormal];
1387  int nextIndex = (((faceNormal + 1) < vertexData.vertexCount) ? (faceNormal + 1) : 0);
1388  Vector2 v2 = vertexData.positions[nextIndex];
1389 
1390  // Check to see if center is within polygon
1391  if (separation < PHYSAC_EPSILON)
1392  {
1393  manifold->contactsCount = 1;
1394  Vector2 normal = Mat2MultiplyVector2(bodyB->shape.transform, vertexData.normals[faceNormal]);
1395  manifold->normal = (Vector2){ -normal.x, -normal.y };
1396  manifold->contacts[0] = (Vector2){ manifold->normal.x*bodyA->shape.radius + bodyA->position.x, manifold->normal.y*bodyA->shape.radius + bodyA->position.y };
1397  manifold->penetration = bodyA->shape.radius;
1398  return;
1399  }
1400 
1401  // Determine which voronoi region of the edge center of circle lies within
1402  float dot1 = MathDot(Vector2Subtract(center, v1), Vector2Subtract(v2, v1));
1403  float dot2 = MathDot(Vector2Subtract(center, v2), Vector2Subtract(v1, v2));
1404  manifold->penetration = bodyA->shape.radius - separation;
1405 
1406  if (dot1 <= 0.0f) // Closest to v1
1407  {
1408  if (DistSqr(center, v1) > bodyA->shape.radius*bodyA->shape.radius) return;
1409 
1410  manifold->contactsCount = 1;
1411  Vector2 normal = Vector2Subtract(v1, center);
1412  normal = Mat2MultiplyVector2(bodyB->shape.transform, normal);
1413  MathNormalize(&normal);
1414  manifold->normal = normal;
1415  v1 = Mat2MultiplyVector2(bodyB->shape.transform, v1);
1416  v1 = Vector2Add(v1, bodyB->position);
1417  manifold->contacts[0] = v1;
1418  }
1419  else if (dot2 <= 0.0f) // Closest to v2
1420  {
1421  if (DistSqr(center, v2) > bodyA->shape.radius*bodyA->shape.radius) return;
1422 
1423  manifold->contactsCount = 1;
1424  Vector2 normal = Vector2Subtract(v2, center);
1425  v2 = Mat2MultiplyVector2(bodyB->shape.transform, v2);
1426  v2 = Vector2Add(v2, bodyB->position);
1427  manifold->contacts[0] = v2;
1428  normal = Mat2MultiplyVector2(bodyB->shape.transform, normal);
1429  MathNormalize(&normal);
1430  manifold->normal = normal;
1431  }
1432  else // Closest to face
1433  {
1434  Vector2 normal = vertexData.normals[faceNormal];
1435 
1436  if (MathDot(Vector2Subtract(center, v1), normal) > bodyA->shape.radius) return;
1437 
1438  normal = Mat2MultiplyVector2(bodyB->shape.transform, normal);
1439  manifold->normal = (Vector2){ -normal.x, -normal.y };
1440  manifold->contacts[0] = (Vector2){ manifold->normal.x*bodyA->shape.radius + bodyA->position.x, manifold->normal.y*bodyA->shape.radius + bodyA->position.y };
1441  manifold->contactsCount = 1;
1442  }
1443 }

1461 // Solves collision between two polygons shape physics bodies
1462 static void SolvePolygonToPolygon(PhysicsManifold manifold)
1463 {
1464  if ((manifold->bodyA == NULL) || (manifold->bodyB == NULL)) return;
1465 
1466  PhysicsShape bodyA = manifold->bodyA->shape;
1467  PhysicsShape bodyB = manifold->bodyB->shape;
1468  manifold->contactsCount = 0;
1469 
1470  // Check for separating axis with A shape's face planes
1471  int faceA = 0;
1472  float penetrationA = FindAxisLeastPenetration(&faceA, bodyA, bodyB);
1473  if (penetrationA >= 0.0f) return;
1474 
1475  // Check for separating axis with B shape's face planes
1476  int faceB = 0;
1477  float penetrationB = FindAxisLeastPenetration(&faceB, bodyB, bodyA);
1478  if (penetrationB >= 0.0f) return;
1479 
1480  int referenceIndex = 0;
1481  bool flip = false; // Always point from A shape to B shape
1482 
1483  PhysicsShape refPoly; // Reference
1484  PhysicsShape incPoly; // Incident
1485 
1486  // Determine which shape contains reference face
1487  if (BiasGreaterThan(penetrationA, penetrationB))
1488  {
1489  refPoly = bodyA;
1490  incPoly = bodyB;
1491  referenceIndex = faceA;
1492  }
1493  else
1494  {
1495  refPoly = bodyB;
1496  incPoly = bodyA;
1497  referenceIndex = faceB;
1498  flip = true;
1499  }
1500 
1501  // World space incident face
1502  Vector2 incidentFace[2];
1503  FindIncidentFace(&incidentFace[0], &incidentFace[1], refPoly, incPoly, referenceIndex);
1504 
1505  // Setup reference face vertices
1506  PolygonData refData = refPoly.vertexData;
1507  Vector2 v1 = refData.positions[referenceIndex];
1508  referenceIndex = (((referenceIndex + 1) < refData.vertexCount) ? (referenceIndex + 1) : 0);
1509  Vector2 v2 = refData.positions[referenceIndex];
1510 
1511  // Transform vertices to world space
1512  v1 = Mat2MultiplyVector2(refPoly.transform, v1);
1513  v1 = Vector2Add(v1, refPoly.body->position);
1514  v2 = Mat2MultiplyVector2(refPoly.transform, v2);
1515  v2 = Vector2Add(v2, refPoly.body->position);
1516 
1517  // Calculate reference face side normal in world space
1518  Vector2 sidePlaneNormal = Vector2Subtract(v2, v1);
1519  MathNormalize(&sidePlaneNormal);
1520 
1521  // Orthogonalize
1522  Vector2 refFaceNormal = { sidePlaneNormal.y, -sidePlaneNormal.x };
1523  float refC = MathDot(refFaceNormal, v1);
1524  float negSide = MathDot(sidePlaneNormal, v1)*-1;
1525  float posSide = MathDot(sidePlaneNormal, v2);
1526 
1527  // Clip incident face to reference face side planes (due to floating point error, possible to not have required points
1528  if (Clip((Vector2){ -sidePlaneNormal.x, -sidePlaneNormal.y }, negSide, &incidentFace[0], &incidentFace[1]) < 2) return;
1529  if (Clip(sidePlaneNormal, posSide, &incidentFace[0], &incidentFace[1]) < 2) return;
1530 
1531  // Flip normal if required
1532  manifold->normal = (flip ? (Vector2){ -refFaceNormal.x, -refFaceNormal.y } : refFaceNormal);
1533 
1534  // Keep points behind reference face
1535  int currentPoint = 0; // Clipped points behind reference face
1536  float separation = MathDot(refFaceNormal, incidentFace[0]) - refC;
1537  if (separation <= 0.0f)
1538  {
1539  manifold->contacts[currentPoint] = incidentFace[0];
1540  manifold->penetration = -separation;
1541  currentPoint++;
1542  }
1543  else manifold->penetration = 0.0f;
1544 
1545  separation = MathDot(refFaceNormal, incidentFace[1]) - refC;
1546 
1547  if (separation <= 0.0f)
1548  {
1549  manifold->contacts[currentPoint] = incidentFace[1];
1550  manifold->penetration += -separation;
1551  currentPoint++;
1552 
1553  // Calculate total penetration average
1554  manifold->penetration /= currentPoint;
1555  }
1556 
1557  manifold->contactsCount = currentPoint;
1558 }

1577 // Initializes physics manifolds to solve collisions
1578 static void InitializePhysicsManifolds(PhysicsManifold manifold)
1579 {
1580  PhysicsBody bodyA = manifold->bodyA;
1581  PhysicsBody bodyB = manifold->bodyB;
1582 
1583  if ((bodyA == NULL) || (bodyB == NULL)) return;
1584 
1585  // Calculate average restitution, static and dynamic friction
1586  manifold->restitution = sqrtf(bodyA->restitution*bodyB->restitution);
1587  manifold->staticFriction = sqrtf(bodyA->staticFriction*bodyB->staticFriction);
1588  manifold->dynamicFriction = sqrtf(bodyA->dynamicFriction*bodyB->dynamicFriction);
1589 
1590  for (int i = 0; i < manifold->contactsCount; i++)
1591  {
1592  // Caculate radius from center of mass to contact

1597  Vector2 crossB = MathCross(bodyB->angularVelocity, radiusB);
1598 
1599  Vector2 radiusV = { 0.0f, 0.0f };
1600  radiusV.x = bodyB->velocity.x + crossB.x - bodyA->velocity.x - crossA.x;
1601  radiusV.y = bodyB->velocity.y + crossB.y - bodyA->velocity.y - crossA.y;
1602 
1603  // Determine if we should perform a resting collision or not;
1604  // The idea is if the only thing moving this object is gravity, then the collision should be performed without any restitution
1605  if (MathLenSqr(radiusV) < (MathLenSqr((Vector2){ gravityForce.x*deltaTime/1000, gravityForce.y*deltaTime/1000 }) + PHYSAC_EPSILON)) manifold->restitution = 0;
1606  }
1607 }

1609 // Integrates physics collisions impulses to solve collisions
1610 static void IntegratePhysicsImpulses(PhysicsManifold manifold)
1611 {
1612  PhysicsBody bodyA = manifold->bodyA;
1613  PhysicsBody bodyB = manifold->bodyB;
1614 
1615  if ((bodyA == NULL) || (bodyB == NULL)) return;
1616 
1617  // Early out and positional correct if both objects have infinite mass
1618  if (fabs(bodyA->inverseMass + bodyB->inverseMass) <= PHYSAC_EPSILON)
1619  {
1620  bodyA->velocity = PHYSAC_VECTOR_ZERO;
1621  bodyB->velocity = PHYSAC_VECTOR_ZERO;
1622  return;
1623  }
1624 
1625  for (int i = 0; i < manifold->contactsCount; i++)
1626  {
1627  // Calculate radius from center of mass to contact
1628  Vector2 radiusA = Vector2Subtract(manifold->contacts[i], bodyA->position);
1629  Vector2 radiusB = Vector2Subtract(manifold->contacts[i], bodyB->position);
1630 
1631  // Calculate relative velocity
1632  Vector2 radiusV = { 0.0f, 0.0f };
1633  radiusV.x = bodyB->velocity.x + MathCross(bodyB->angularVelocity, radiusB).x - bodyA->velocity.x - MathCross(bodyA->angularVelocity, radiusA).x;
1634  radiusV.y = bodyB->velocity.y + MathCross(bodyB->angularVelocity, radiusB).y - bodyA->velocity.y - MathCross(bodyA->angularVelocity, radiusA).y;
1635 
1636  // Relative velocity along the normal
1637  float contactVelocity = MathDot(radiusV, manifold->normal);
1638 
1639  // Do not resolve if velocities are separating
1640  if (contactVelocity > 0.0f) return;
1641 
1642  float raCrossN = MathCrossVector2(radiusA, manifold->normal);
1643  float rbCrossN = MathCrossVector2(radiusB, manifold->normal);
1644 
1645  float inverseMassSum = bodyA->inverseMass + bodyB->inverseMass + (raCrossN*raCrossN)*bodyA->inverseInertia + (rbCrossN*rbCrossN)*bodyB->inverseInertia;
1646 
1647  // Calculate impulse scalar value
1648  float impulse = -(1.0f + manifold->restitution)*contactVelocity;
1649  impulse /= inverseMassSum;
1650  impulse /= (float)manifold->contactsCount;
1651 
1652  // Apply impulse to each physics body
1653  Vector2 impulseV = { manifold->normal.x*impulse, manifold->normal.y*impulse };
1654 
1655  if (bodyA->enabled)
1656  {
1657  bodyA->velocity.x += bodyA->inverseMass*(-impulseV.x);
1658  bodyA->velocity.y += bodyA->inverseMass*(-impulseV.y);
1659  if (!bodyA->freezeOrient) bodyA->angularVelocity += bodyA->inverseInertia*MathCrossVector2(radiusA, (Vector2){ -impulseV.x, -impulseV.y });
1660  }
1661 
1662  if (bodyB->enabled)
1663  {
1664  bodyB->velocity.x += bodyB->inverseMass*(impulseV.x);
1665  bodyB->velocity.y += bodyB->inverseMass*(impulseV.y);
1666  if (!bodyB->freezeOrient) bodyB->angularVelocity += bodyB->inverseInertia*MathCrossVector2(radiusB, impulseV);
1667  }
1668 
1669  // Apply friction impulse to each physics body
1670  radiusV.x = bodyB->velocity.x + MathCross(bodyB->angularVelocity, radiusB).x - bodyA->velocity.x - MathCross(bodyA->angularVelocity, radiusA).x;
1671  radiusV.y = bodyB->velocity.y + MathCross(bodyB->angularVelocity, radiusB).y - bodyA->velocity.y - MathCross(bodyA->angularVelocity, radiusA).y;
1672 
1673  Vector2 tangent = { radiusV.x - (manifold->normal.x*MathDot(radiusV, manifold->normal)), radiusV.y - (manifold->normal.y*MathDot(radiusV, manifold->normal)) };
1674  MathNormalize(&tangent);
1675 
1676  // Calculate impulse tangent magnitude
1677  float impulseTangent = -MathDot(radiusV, tangent);
1678  impulseTangent /= inverseMassSum;
1679  impulseTangent /= (float)manifold->contactsCount;
1680 
1681  float absImpulseTangent = fabs(impulseTangent);
1682 
1683  // Don't apply tiny friction impulses
1684  if (absImpulseTangent <= PHYSAC_EPSILON) return;
1685 
1686  // Apply coulumb's law
1687  Vector2 tangentImpulse = { 0.0f, 0.0f };
1688  if (absImpulseTangent < impulse*manifold->staticFriction) tangentImpulse = (Vector2){ tangent.x*impulseTangent, tangent.y*impulseTangent };
1689  else tangentImpulse = (Vector2){ tangent.x*-impulse*manifold->dynamicFriction, tangent.y*-impulse*manifold->dynamicFriction };
1690 
1691  // Apply friction impulse
1692  if (bodyA->enabled)
1693  {
1694  bodyA->velocity.x += bodyA->inverseMass*(-tangentImpulse.x);
1695  bodyA->velocity.y += bodyA->inverseMass*(-tangentImpulse.y);
1696 
1697  if (!bodyA->freezeOrient) bodyA->angularVelocity += bodyA->inverseInertia*MathCrossVector2(radiusA, (Vector2){ -tangentImpulse.x, -tangentImpulse.y });
1698  }
1699 
1700  if (bodyB->enabled)
1701  {
1702  bodyB->velocity.x += bodyB->inverseMass*(tangentImpulse.x);
1703  bodyB->velocity.y += bodyB->inverseMass*(tangentImpulse.y);
1704 
1705  if (!bodyB->freezeOrient) bodyB->angularVelocity += bodyB->inverseInertia*MathCrossVector2(radiusB, tangentImpulse);
1706  }
1707  }
1708 }

1724 // Corrects physics bodies positions based on manifolds collision information
