189  void PollInput()
190  {
191  // Process Messages
192  {
193  // Loop over pending messages
194  while (UWPHasMessages())
195  {
196  // Get the message
197  auto msg = UWPGetMessage();
198 
199  // Carry out the command
200  switch(msg->type)
201  {
202  case UWP_MSG_SHOW_MOUSE: // Do the same thing because of how UWP works...
203  case UWP_MSG_UNLOCK_MOUSE:
204  {
205  CoreWindow::GetForCurrentThread()->PointerCursor = regularCursor;
206  cursorLocked = false;
207  MoveMouse(GetMousePosition());
208  break;
209  }
210  case UWP_MSG_HIDE_MOUSE: // Do the same thing because of how UWP works...
211  case UWP_MSG_LOCK_MOUSE:
212  {
213  CoreWindow::GetForCurrentThread()->PointerCursor = nullptr;
214  cursorLocked = true;
215  break;
216  }
217  case UWP_MSG_SET_MOUSE_LOCATION:
218  {
219  MoveMouse(msg->paramVector0);
220  break;
221  }
222  }
223 
224  // Delete the message
225  DeleteUWPMessage(msg);
226  }
227  }
228 
229  // Process Keyboard
230  {
231  for (int k = 0x08; k < 0xA6; k++) {
232  auto state = CoreWindow::GetForCurrentThread()->GetKeyState((Windows::System::VirtualKey) k);
233 
234 #ifdef HOLDHACK
235  // Super hacky way of waiting three frames to see if we are ready to register the key as deregistered
236  // This will wait an entire 4 frames before deregistering the key, this makes sure that the key is not flickering
237  if (KeyboardStateHack[k] == 2)
238  {
239  if ((state & CoreVirtualKeyStates::None) == CoreVirtualKeyStates::None)
240  {
241  KeyboardStateHack[k] = 3;
242  }
243  }
244  else if (KeyboardStateHack[k] == 3)
245  {
246  if ((state & CoreVirtualKeyStates::None) == CoreVirtualKeyStates::None)
247  {
248  KeyboardStateHack[k] = 4;
249  }
250  }
251  else if (KeyboardStateHack[k] == 4)
252  {
253  if ((state & CoreVirtualKeyStates::None) == CoreVirtualKeyStates::None)
254  {
255  //Reset key...
256  KeyboardStateHack[k] = 0;
257 
258  //Tell core
259  RegisterKey(k, 0);
260  }
261  }
262 #endif
263  // Left and right alt, KeyUp and KeyDown are not called for it
264  // No need to hack because this is not a character
265 
266  // TODO: Maybe do all other key registrations like this, no more key events?
267 
268  if (k == 0xA4 || k == 0xA5)
269  {
270  if ((state & CoreVirtualKeyStates::Down) == CoreVirtualKeyStates::Down)
271  {
272  RegisterKey(k, 1);
273  }
274  else
275  {
276  RegisterKey(k, 0);
277  }
278  }
279  }
280  }
281 
282  // Process Mouse
283  {
284 
285  if (CurrentPointerID > -1)
286  {
287  auto point = PointerPoint::GetCurrentPoint(CurrentPointerID);
288  auto props = point->Properties;
289 
290  if (props->IsLeftButtonPressed)
291  {
292  RegisterClick(MOUSE_LEFT_BUTTON, 1);
293  }
294  else
295  {
296  RegisterClick(MOUSE_LEFT_BUTTON, 0);
297  }
298 
299  if (props->IsRightButtonPressed)
300  {
301  RegisterClick(MOUSE_RIGHT_BUTTON, 1);
302  }
303  else
304  {
305  RegisterClick(MOUSE_RIGHT_BUTTON, 0);
306  }
307 
308  if (props->IsMiddleButtonPressed)
309  {
310  RegisterClick(MOUSE_MIDDLE_BUTTON, 1);
311  }
312  else
313  {
314  RegisterClick(MOUSE_MIDDLE_BUTTON, 0);
315  }
316  }

318  CoreWindow ^window = CoreWindow::GetForCurrentThread();
319 
320  if (cursorLocked)
321  {
322  // Track cursor movement delta, recenter it on the client
323  auto curMousePos = GetMousePosition();
324 
325  auto x = curMousePos.x + mouseDelta.x;
326  auto y = curMousePos.y + mouseDelta.y;
327 
328  UpdateMousePosition({ x, y });
329 
330  // Why we're not using UWPSetMousePosition here...
331  // UWPSetMousePosition changes the "mousePosition" variable to match where the cursor actually is.
332  // Our cursor is locked to the middle of screen, and we don't want that reflected in "mousePosition"
333  Vector2 centerClient = { (float)(GetScreenWidth() / 2), (float)(GetScreenHeight() / 2) };
334  window->PointerPosition = Point(centerClient.x + window->Bounds.X, centerClient.y + window->Bounds.Y);
335  }
336  else
337  {
338  // Record the cursor's position relative to the client
339  auto x = window->PointerPosition.X - window->Bounds.X;
340  auto y = window->PointerPosition.Y - window->Bounds.Y;
341 
342  UpdateMousePosition({ x, y });
343  }
344 
345  mouseDelta = { 0 ,0 };
346  }

348  // Process Gamepads
349  {
350  // Check if gamepads are ready
351  for (int i = 0; i < MAX_GAMEPADS; i++)
352  {
353  // HACK: UWP keeps a contiguous list of gamepads. For the interest of time I'm just doing a 1:1 mapping of
354  // connected gamepads with their spot in the list, but this has serious robustness problems
355  // e.g. player 1, 2, and 3 are playing a game - if player2 disconnects, p3's controller would now be mapped to p2's character since p3 is now second in the list.
356 
357  UWPMessage* msg = CreateUWPMessage();
358  msg->type = UWP_MSG_SET_GAMEPAD_ACTIVE;
359  msg->paramInt0 = i;
360  msg->paramBool0 = i < Gamepad::Gamepads->Size;
361  UWPSendMessage(msg);
362  }

365  for (int i = 0; i < MAX_GAMEPADS; i++)
366  {

372

375  RegisterGamepadButton(i, GAMEPAD_BUTTON_RIGHT_FACE_RIGHT, ((reading.Buttons & GamepadButtons::B) == GamepadButtons::B));
376  RegisterGamepadButton(i, GAMEPAD_BUTTON_RIGHT_FACE_LEFT, ((reading.Buttons & GamepadButtons::X) == GamepadButtons::X));
377  RegisterGamepadButton(i, GAMEPAD_BUTTON_RIGHT_FACE_UP, ((reading.Buttons & GamepadButtons::Y) == GamepadButtons::Y));
378 
379  RegisterGamepadButton(i, GAMEPAD_BUTTON_LEFT_TRIGGER_1, ((reading.Buttons & GamepadButtons::LeftShoulder) == GamepadButtons::LeftShoulder));
380  RegisterGamepadButton(i, GAMEPAD_BUTTON_RIGHT_TRIGGER_1, ((reading.Buttons & GamepadButtons::RightShoulder) == GamepadButtons::RightShoulder));
381 
382  RegisterGamepadButton(i, GAMEPAD_BUTTON_MIDDLE_LEFT, ((reading.Buttons & GamepadButtons::View) == GamepadButtons::View)); // Changed for XB1 Controller
383  RegisterGamepadButton(i, GAMEPAD_BUTTON_MIDDLE_RIGHT, ((reading.Buttons & GamepadButtons::Menu) == GamepadButtons::Menu)); // Changed for XB1 Controller
384 
385  RegisterGamepadButton(i, GAMEPAD_BUTTON_LEFT_FACE_UP, ((reading.Buttons & GamepadButtons::DPadUp) == GamepadButtons::DPadUp));
386  RegisterGamepadButton(i, GAMEPAD_BUTTON_LEFT_FACE_RIGHT, ((reading.Buttons & GamepadButtons::DPadRight) == GamepadButtons::DPadRight));
387  RegisterGamepadButton(i, GAMEPAD_BUTTON_LEFT_FACE_DOWN, ((reading.Buttons & GamepadButtons::DPadDown) == GamepadButtons::DPadDown));
388  RegisterGamepadButton(i, GAMEPAD_BUTTON_LEFT_FACE_LEFT, ((reading.Buttons & GamepadButtons::DPadLeft) == GamepadButtons::DPadLeft));
389  RegisterGamepadButton(i, GAMEPAD_BUTTON_MIDDLE, false); // Home button not supported by UWP

393  RegisterGamepadAxis(i, GAMEPAD_AXIS_LEFT_Y, (float)reading.LeftThumbstickY);
