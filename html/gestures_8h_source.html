249 // Process gesture event and translate it into gestures
250 void ProcessGestureEvent(GestureEvent event)
251 {
252  // Reset required variables
253  pointCount = event.pointCount; // Required on UpdateGestures()
254 
255  if (pointCount < 2)
256  {
257  if (event.touchAction == TOUCH_DOWN)
258  {
259  tapCounter++; // Tap counter
260 
261  // Detect GESTURE_DOUBLE_TAP
262  if ((currentGesture == GESTURE_NONE) && (tapCounter >= 2) && ((GetCurrentTime() - eventTime) < TAP_TIMEOUT) && (Vector2Distance(touchDownPosition, event.position[0]) < DOUBLETAP_RANGE))
263  {
264  currentGesture = GESTURE_DOUBLETAP;
265  tapCounter = 0;
266  }
267  else // Detect GESTURE_TAP
268  {
269  tapCounter = 1;
270  currentGesture = GESTURE_TAP;
271  }
272 
273  touchDownPosition = event.position[0];
274  touchDownDragPosition = event.position[0];
275 
276  touchUpPosition = touchDownPosition;
277  eventTime = GetCurrentTime();
278 
279  firstTouchId = event.pointerId[0];
280 
281  dragVector = (Vector2){ 0.0f, 0.0f };
282  }
283  else if (event.touchAction == TOUCH_UP)
284  {
285  if (currentGesture == GESTURE_DRAG) touchUpPosition = event.position[0];
286 
287  // NOTE: dragIntensity dependend on the resolution of the screen
288  dragDistance = Vector2Distance(touchDownPosition, touchUpPosition);
289  dragIntensity = dragDistance/(float)((GetCurrentTime() - swipeTime));
290 
291  startMoving = false;
292 
293  // Detect GESTURE_SWIPE
294  if ((dragIntensity > FORCE_TO_SWIPE) && (firstTouchId == event.pointerId[0]))
295  {
296  // NOTE: Angle should be inverted in Y
297  dragAngle = 360.0f - Vector2Angle(touchDownPosition, touchUpPosition);
298 
299  if ((dragAngle < 30) || (dragAngle > 330)) currentGesture = GESTURE_SWIPE_RIGHT; // Right
300  else if ((dragAngle > 30) && (dragAngle < 120)) currentGesture = GESTURE_SWIPE_UP; // Up
301  else if ((dragAngle > 120) && (dragAngle < 210)) currentGesture = GESTURE_SWIPE_LEFT; // Left
302  else if ((dragAngle > 210) && (dragAngle < 300)) currentGesture = GESTURE_SWIPE_DOWN; // Down
303  else currentGesture = GESTURE_NONE;
304  }
305  else
306  {
307  dragDistance = 0.0f;
308  dragIntensity = 0.0f;
309  dragAngle = 0.0f;
310 
311  currentGesture = GESTURE_NONE;
312  }
313 
314  touchDownDragPosition = (Vector2){ 0.0f, 0.0f };
315  pointCount = 0;
316  }
317  else if (event.touchAction == TOUCH_MOVE)
318  {
319  if (currentGesture == GESTURE_DRAG) eventTime = GetCurrentTime();
320 
321  if (!startMoving)
322  {
323  swipeTime = GetCurrentTime();
324  startMoving = true;
325  }
326 
327  moveDownPosition = event.position[0];
328 
329  if (currentGesture == GESTURE_HOLD)
330  {
331  if (resetHold) touchDownPosition = event.position[0];
332 
333  resetHold = false;
334 
335  // Detect GESTURE_DRAG
336  if (Vector2Distance(touchDownPosition, moveDownPosition) >= MINIMUM_DRAG)
337  {
338  eventTime = GetCurrentTime();
339  currentGesture = GESTURE_DRAG;
340  }
341  }
342 
343  dragVector.x = moveDownPosition.x - touchDownDragPosition.x;
344  dragVector.y = moveDownPosition.y - touchDownDragPosition.y;
345  }
346  }
347  else // Two touch points
348  {
349  if (event.touchAction == TOUCH_DOWN)
350  {
351  touchDownPosition = event.position[0];
352  touchDownPosition2 = event.position[1];
353 
354  //pinchDistance = Vector2Distance(touchDownPosition, touchDownPosition2);
355 
356  pinchVector.x = touchDownPosition2.x - touchDownPosition.x;
357  pinchVector.y = touchDownPosition2.y - touchDownPosition.y;
358 
359  currentGesture = GESTURE_HOLD;
360  timeHold = GetCurrentTime();
361  }
362  else if (event.touchAction == TOUCH_MOVE)
363  {
364  pinchDistance = Vector2Distance(moveDownPosition, moveDownPosition2);
365 
366  touchDownPosition = moveDownPosition;
367  touchDownPosition2 = moveDownPosition2;
368 
369  moveDownPosition = event.position[0];
370  moveDownPosition2 = event.position[1];
371 
372  pinchVector.x = moveDownPosition2.x - moveDownPosition.x;
373  pinchVector.y = moveDownPosition2.y - moveDownPosition.y;
374 
375  if ((Vector2Distance(touchDownPosition, moveDownPosition) >= MINIMUM_PINCH) || (Vector2Distance(touchDownPosition2, moveDownPosition2) >= MINIMUM_PINCH))
376  {
377  if ((Vector2Distance(moveDownPosition, moveDownPosition2) - pinchDistance) < 0) currentGesture = GESTURE_PINCH_IN;
378  else currentGesture = GESTURE_PINCH_OUT;
379  }
380  else
381  {
382  currentGesture = GESTURE_HOLD;
383  timeHold = GetCurrentTime();
384  }
385 
386  // NOTE: Angle should be inverted in Y
387  pinchAngle = 360.0f - Vector2Angle(moveDownPosition, moveDownPosition2);
388  }
389  else if (event.touchAction == TOUCH_UP)
390  {
391  pinchDistance = 0.0f;
392  pinchAngle = 0.0f;
393  pinchVector = (Vector2){ 0.0f, 0.0f };
394  pointCount = 0;
395 
396  currentGesture = GESTURE_NONE;
397  }
398  }
399 }

401 // Update gestures detected (must be called every frame)
402 void UpdateGestures(void)
403 {
404  // NOTE: Gestures are processed through system callbacks on touch events
405 
406  // Detect GESTURE_HOLD
407  if (((currentGesture == GESTURE_TAP) || (currentGesture == GESTURE_DOUBLETAP)) && (pointCount < 2))
408  {
409  currentGesture = GESTURE_HOLD;
410  timeHold = GetCurrentTime();
411  }
412 
413  if (((GetCurrentTime() - eventTime) > TAP_TIMEOUT) && (currentGesture == GESTURE_DRAG) && (pointCount < 2))
414  {
415  currentGesture = GESTURE_HOLD;
416  timeHold = GetCurrentTime();
417  resetHold = true;
418  }
419 
420  // Detect GESTURE_NONE
421  if ((currentGesture == GESTURE_SWIPE_RIGHT) || (currentGesture == GESTURE_SWIPE_UP) || (currentGesture == GESTURE_SWIPE_LEFT) || (currentGesture == GESTURE_SWIPE_DOWN))
422  {
423  currentGesture = GESTURE_NONE;
424  }
425 }

506  float result;
