245 void *MemPoolAlloc(MemPool *const mempool, const size_t size)
246 {
247  if ((mempool == NULL) || (size == 0UL) || (size > mempool->stack.size)) return NULL;
248  else
249  {
250  MemNode *new_mem = NULL;
251  const size_t ALLOC_SIZE = __AlignSize(size + sizeof *new_mem, sizeof(intptr_t));
252  const size_t BUCKET_INDEX = (ALLOC_SIZE >> MEMPOOL_BUCKET_BITS) - 1;
253 
254  // If the size is small enough, let's check if our buckets has a fitting memory block.
255  if (BUCKET_INDEX < MEMPOOL_BUCKET_SIZE && mempool->buckets[BUCKET_INDEX] != NULL && mempool->buckets[BUCKET_INDEX]->size >= ALLOC_SIZE)
256  {
257  new_mem = mempool->buckets[BUCKET_INDEX];
258  mempool->buckets[BUCKET_INDEX] = mempool->buckets[BUCKET_INDEX]->next;
259  if( mempool->buckets[BUCKET_INDEX] != NULL )
260  mempool->buckets[BUCKET_INDEX]->prev = NULL;
261  }
262  else if (mempool->freeList.head != NULL)
263  {
264  const size_t MEM_SPLIT_THRESHOLD = 16;
265 
266  // If the freelist is valid, let's allocate FROM the freelist then!
267  for (MemNode *inode = mempool->freeList.head; inode != NULL; inode = inode->next)
268  {
269  if (inode->size < ALLOC_SIZE) continue;
270  else if (inode->size <= (ALLOC_SIZE + MEM_SPLIT_THRESHOLD))
271  {
272  // Close in size - reduce fragmentation by not splitting.
273  new_mem = inode;
274  (inode->prev != NULL)? (inode->prev->next = inode->next) : (mempool->freeList.head = inode->next);
275  (inode->next != NULL)? (inode->next->prev = inode->prev) : (mempool->freeList.tail = inode->prev);
276 
277  if (mempool->freeList.head != NULL) mempool->freeList.head->prev = NULL;
278  else mempool->freeList.tail = NULL;
279 
280  if (mempool->freeList.tail != NULL) mempool->freeList.tail->next = NULL;
281  mempool->freeList.len--;
282  break;
283  }
284  else
285  {
286  // Split the memory chunk.
287  new_mem = (MemNode *)((uint8_t *)inode + (inode->size - ALLOC_SIZE));
288  inode->size -= ALLOC_SIZE;
289  new_mem->size = ALLOC_SIZE;
290  break;
291  }
292  }
293  }
294 
295  if (new_mem == NULL)
296  {
297  // not enough memory to support the size!
298  if ((mempool->stack.base - ALLOC_SIZE) < mempool->stack.mem) return NULL;
299  else
300  {
301  // Couldn't allocate from a freelist, allocate from available mempool.
302  // Subtract allocation size from the mempool.
303  mempool->stack.base -= ALLOC_SIZE;
304 
305  // Use the available mempool space as the new node.
306  new_mem = (MemNode *)mempool->stack.base;
307  new_mem->size = ALLOC_SIZE;
308  }
309  }
310 
311  // Visual of the allocation block.
312  // --------------
313  // | mem size | lowest addr of block
314  // | next node | 12 byte (32-bit) header
315  // | prev node | 24 byte (64-bit) header
316  // --------------
317  // | alloc'd |
318  // | memory |
319  // | space | highest addr of block
320  // --------------
321  new_mem->next = new_mem->prev = NULL;
322  uint8_t *const final_mem = (uint8_t *)new_mem + sizeof *new_mem;
323  return memset(final_mem, 0, new_mem->size - sizeof *new_mem);
324  }
325 }

327 void *MemPoolRealloc(MemPool *const restrict mempool, void *ptr, const size_t size)
328 {
329  if ((mempool == NULL) || (size > mempool->stack.size)) return NULL;
330  // NULL ptr should make this work like regular Allocation.
331  else if (ptr == NULL) return MemPoolAlloc(mempool, size);
332  else if ((uintptr_t)ptr - sizeof(MemNode) < (uintptr_t)mempool->stack.mem) return NULL;
333  else
334  {
335  MemNode *const node = (MemNode *)((uint8_t *)ptr - sizeof *node);
336  const size_t NODE_SIZE = sizeof *node;
337  uint8_t *const resized_block = MemPoolAlloc(mempool, size);
338 
339  if (resized_block == NULL) return NULL;
340  else
341  {
342  MemNode *const resized = (MemNode *)(resized_block - sizeof *resized);
343  memmove(resized_block, ptr, (node->size > resized->size)? (resized->size - NODE_SIZE) : (node->size - NODE_SIZE));
344  MemPoolFree(mempool, ptr);
345  return resized_block;
346  }
347  }

350 void MemPoolFree(MemPool *const restrict mempool, void *ptr)
351 {
352  if ((mempool == NULL) || (ptr == NULL) || ((uintptr_t)ptr - sizeof(MemNode) < (uintptr_t)mempool->stack.mem)) return;
353  else
354  {
355  // Behind the actual pointer data is the allocation info.
356  MemNode *const mem_node = (MemNode *)((uint8_t *)ptr - sizeof *mem_node);
357  const size_t BUCKET_INDEX = (mem_node->size >> MEMPOOL_BUCKET_BITS) - 1;
358 
359  // Make sure the pointer data is valid.
360  if (((uintptr_t)mem_node < (uintptr_t)mempool->stack.base) ||
361  (((uintptr_t)mem_node - (uintptr_t)mempool->stack.mem) > mempool->stack.size) ||
362  (mem_node->size == 0UL) ||
363  (mem_node->size > mempool->stack.size)) return;
364  // If the mem_node is right at the stack base ptr, then add it to the stack.
365  else if ((uintptr_t)mem_node == (uintptr_t)mempool->stack.base)
366  {
367  mempool->stack.base += mem_node->size;
368  }
369  // attempted stack merge failed, try to place it into the memnode buckets
370  else if (BUCKET_INDEX < MEMPOOL_BUCKET_SIZE)
371  {
372  if (mempool->buckets[BUCKET_INDEX] == NULL) mempool->buckets[BUCKET_INDEX] = mem_node;
373  else
374  {
375  for (MemNode *n = mempool->buckets[BUCKET_INDEX]; n != NULL; n = n->next) if( n==mem_node ) return;
376  mempool->buckets[BUCKET_INDEX]->prev = mem_node;
377  mem_node->next = mempool->buckets[BUCKET_INDEX];
378  mempool->buckets[BUCKET_INDEX] = mem_node;
379  }
380  }
381  // Otherwise, we add it to the free list.
382  // We also check if the freelist already has the pointer so we can prevent double frees.
383  else /*if ((mempool->freeList.len == 0UL) || ((uintptr_t)mempool->freeList.head >= (uintptr_t)mempool->stack.mem && (uintptr_t)mempool->freeList.head - (uintptr_t)mempool->stack.mem < mempool->stack.size))*/
384  {
385  for (MemNode *n = mempool->freeList.head; n != NULL; n = n->next) if (n == mem_node) return;
386 
387  // This code insertion sorts where largest size is last.
388  if (mempool->freeList.head == NULL)
389  {
390  mempool->freeList.head = mempool->freeList.tail = mem_node;
391  mempool->freeList.len++;
392  }
393  else if (mempool->freeList.head->size >= mem_node->size)
394  {
395  mem_node->next = mempool->freeList.head;
396  mem_node->next->prev = mem_node;
397  mempool->freeList.head = mem_node;
398  mempool->freeList.len++;
399  }
400  else //if (mempool->freeList.tail->size <= mem_node->size)
401  {
402  mem_node->prev = mempool->freeList.tail;
403  mempool->freeList.tail->next = mem_node;
404  mempool->freeList.tail = mem_node;
405  mempool->freeList.len++;
406  }
407 
408  if (mempool->freeList.autoDefrag && (mempool->freeList.maxNodes != 0UL) && (mempool->freeList.len > mempool->freeList.maxNodes)) MemPoolDefrag(mempool);
409  }
410  }
411 }

443 bool MemPoolDefrag(MemPool *const mempool)
444 {
445  if (mempool == NULL) return false;
446  else
447  {
448  // If the memory pool has been entirely released, fully defrag it.
449  if (mempool->stack.size == GetMemPoolFreeMemory(*mempool))
450  {
451  MemPoolReset(mempool);
452  return true;
453  }
454  else
455  {
456  for (size_t i=0; i<MEMPOOL_BUCKET_SIZE; i++)
457  {
458  while (mempool->buckets[i] != NULL)
459  {
460  if ((uintptr_t)mempool->buckets[i] == (uintptr_t)mempool->stack.base)
461  {
462  mempool->stack.base += mempool->buckets[i]->size;
463  mempool->buckets[i]->size = 0;
464  mempool->buckets[i] = mempool->buckets[i]->next;
465  if (mempool->buckets[i] != NULL) mempool->buckets[i]->prev = NULL;
466  }
467  else break;
468  }
469  }
470 
471  const size_t PRE_DEFRAG_LEN = mempool->freeList.len;
472  MemNode **node = &mempool->freeList.head;
473 
474  while (*node != NULL)
475  {
476  if ((uintptr_t)*node == (uintptr_t)mempool->stack.base)
477  {
478  // If node is right at the stack, merge it back into the stack.
479  mempool->stack.base += (*node)->size;
480  (*node)->size = 0UL;
481  ((*node)->prev != NULL)? ((*node)->prev->next = (*node)->next) : (mempool->freeList.head = (*node)->next);
482  ((*node)->next != NULL)? ((*node)->next->prev = (*node)->prev) : (mempool->freeList.tail = (*node)->prev);
483 
484  if (mempool->freeList.head != NULL) mempool->freeList.head->prev = NULL;
485  else mempool->freeList.tail = NULL;
486 
487  if (mempool->freeList.tail != NULL) mempool->freeList.tail->next = NULL;
488  mempool->freeList.len--;
489  node = &mempool->freeList.head;
490  }
491  else if (((uintptr_t)*node + (*node)->size) == (uintptr_t)(*node)->next)
492  {
493  // Next node is at a higher address.
494  (*node)->size += (*node)->next->size;
495  (*node)->next->size = 0UL;
496 
497  // <-[P Curr N]-> <-[P Next N]-> <-[P NextNext N]->
498  //
499  // |--------------------|
500  // <-[P Curr N]-> <-[P Next N]-> [P NextNext N]->
501  if ((*node)->next->next != NULL) (*node)->next->next->prev = *node;
502 
503  // <-[P Curr N]-> <-[P NextNext N]->
504  (*node)->next = (*node)->next->next;
505 
506  mempool->freeList.len--;
507  node = &mempool->freeList.head;
508  }
509  else if ((((uintptr_t)*node + (*node)->size) == (uintptr_t)(*node)->prev) && ((*node)->prev->prev != NULL))
510  {
511  // Prev node is at a higher address.
512  (*node)->size += (*node)->prev->size;
513  (*node)->prev->size = 0UL;
514 
515  // <-[P PrevPrev N]-> <-[P Prev N]-> <-[P Curr N]->
516  //
517  // |--------------------|
518  // <-[P PrevPrev N] <-[P Prev N]-> <-[P Curr N]->
519  (*node)->prev->prev->next = *node;
520 
521  // <-[P PrevPrev N]-> <-[P Curr N]->
522  (*node)->prev = (*node)->prev->prev;
523 
524  mempool->freeList.len--;
525  node = &mempool->freeList.head;
526  }
527  else if ((*node)->prev != NULL && (*node)->next != NULL && (uintptr_t)*node - (*node)->next->size == (uintptr_t)(*node)->next)
528  {
529  // Next node is at a lower address.
530  (*node)->next->size += (*node)->size;
531 
532  (*node)->size = 0UL;
533  (*node)->next->prev = (*node)->prev;
534  (*node)->prev->next = (*node)->next;
535  *node = (*node)->next;
536 
537  mempool->freeList.len--;
538  node = &mempool->freeList.head;
539  }
540  else if ((*node)->prev != NULL && (*node)->next != NULL && (uintptr_t)*node - (*node)->prev->size == (uintptr_t)(*node)->prev)
541  {
542  // Prev node is at a lower address.
543  (*node)->prev->size += (*node)->size;
544 
545  (*node)->size = 0UL;
546  (*node)->next->prev = (*node)->prev;
547  (*node)->prev->next = (*node)->next;
548  *node = (*node)->prev;
549 
550  mempool->freeList.len--;
551  node = &mempool->freeList.head;
552  }
553  else
554  {
555  node = &(*node)->next;
556  }
557  }
558 
559  return PRE_DEFRAG_LEN > mempool->freeList.len;
560  }
561  }

578 ObjPool CreateObjPool(const size_t objsize, const size_t len)
579 {
580  ObjPool objpool = { 0 };
581 
582  if ((len == 0UL) || (objsize == 0UL)) return objpool;
583  else
584  {
585  objpool.objSize = __AlignSize(objsize, sizeof(size_t));
586  objpool.stack.size = objpool.freeBlocks = len;
587  objpool.stack.mem = calloc(objpool.stack.size, objpool.objSize);
588 
589  if (objpool.stack.mem == NULL)
590  {
591  objpool.stack.size = 0UL;
592  return objpool;
593  }
594  else
595  {
596  for (size_t i=0; i<objpool.freeBlocks; i++)
597  {
598  union ObjInfo block = { .byte = &objpool.stack.mem[i*objpool.objSize] };
599  *block.index = i + 1;
600  }
601 
602  objpool.stack.base = objpool.stack.mem;
603  return objpool;
604  }
605  }

608 ObjPool CreateObjPoolFromBuffer(void *const buf, const size_t objsize, const size_t len)
609 {
610  ObjPool objpool = { 0 };
611 
612  // If the object size isn't large enough to align to a size_t, then we can't use it.
613  if ((buf == NULL) || (len == 0UL) || (objsize < sizeof(size_t)) || (objsize*len != __AlignSize(objsize, sizeof(size_t))*len)) return objpool;
614  else
615  {
616  objpool.objSize = __AlignSize(objsize, sizeof(size_t));
617  objpool.stack.size = objpool.freeBlocks = len;
618  objpool.stack.mem = buf;
619 
620  for (size_t i=0; i<objpool.freeBlocks; i++)
621  {

624  }

628  }

641 void *ObjPoolAlloc(ObjPool *const objpool)
642 {
643  if (objpool == NULL) return NULL;
644  else
645  {
646  if (objpool->freeBlocks > 0UL)
647  {
648  // For first allocation, head points to the very first index.
649  // Head = &pool[0];
650  // ret = Head == ret = &pool[0];
651  union ObjInfo ret = { .byte = objpool->stack.base };
652  objpool->freeBlocks--;
653 
654  // after allocating, we set head to the address of the index that *Head holds.
655  // Head = &pool[*Head * pool.objsize];
656  objpool->stack.base = (objpool->freeBlocks != 0UL)? objpool->stack.mem + (*ret.index*objpool->objSize) : NULL;
657  memset(ret.byte, 0, objpool->objSize);
658  return ret.byte;
659  }
660  else return NULL;
661  }
662 }

664 void ObjPoolFree(ObjPool *const restrict objpool, void *ptr)
665 {
666  union ObjInfo p = { .byte = ptr };
667  if ((objpool == NULL) || (ptr == NULL) || (p.byte < objpool->stack.mem) || (p.byte > objpool->stack.mem + objpool->stack.size*objpool->objSize)) return;
668  else
669  {
670  // When we free our pointer, we recycle the pointer space to store the previous index and then we push it as our new head.
671  // *p = index of Head in relation to the buffer;
672  // Head = p;
673  *p.index = (objpool->stack.base != NULL)? (objpool->stack.base - objpool->stack.mem)/objpool->objSize : objpool->stack.size;
674  objpool->stack.base = p.byte;
675  objpool->freeBlocks++;
676  }
677 }
